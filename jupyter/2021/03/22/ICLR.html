df = pd.DataFrame(columns=['ID', 'leaf_rust', 'stem_rust', 'healthy_wheat'])

with learn.no_bar():
    for image in fns:
        file_id = image.name.split(".")[0]
        _,_,pred = learn.predict(image)
        new_row = { 'ID':file_id, 'leaf_rust':pred[2].item(), 'stem_rust':pred[1].item(), 'healthy_wheat':pred[0].item() }
        df.loc[file_id] = new_row

df.set_index('ID', inplace=True)

df = df.round(decimals=2)
